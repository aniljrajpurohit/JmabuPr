;EXPERIMENT 1:WRITE AN ALP TO INTERFACE DAC0808 AND GENERATE SQUARE WAVE
 ;NAME:ANKIT PATEL
 ;ROLL N0:312027
 ;BATCH:B1
 ;DATE OF PERFORMANCE:1/8/2018
 
    ORG 000H
    MOV A,#00H
 L1:MOV P1,A
    ACALL DELAY
    MOV P1,#0FFH
    ACALL DELAY
    SJMP L1
 
DELAY:
    MOV R1,#250d
L2: DJNZ R1,L2
    RET
    END
