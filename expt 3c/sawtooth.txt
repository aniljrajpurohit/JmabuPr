 ;EXPERIMENT 2:WRITE AN ALP TO INTERFACE DAC0808 AND GENERATE SAWTOOTH WAVEFORM
 ;NAME:ANKIT PATEL
 ;ROLL N0:312027
 ;BATCH:B1
 ;DATE OF PERFORMANCE:1/8/2018
 
    ORG 000H
    MOV A,#00H ;INITIALIZING INTITIAL POSITION OF SAWTOOTH WAVE
 L1:MOV P1,A   ;LOAD DATA OF A INTO PORT 0
    INC A      ;INCREMENT STEP OF WAVE
    SJMP L1    ;CONTINUOUS WAVEFORM
    END
